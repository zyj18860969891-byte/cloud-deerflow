.\start-dev-services.ps1                    # 推荐：启动所有服务

# 后端开发
cd backend
.\.venv\Scripts\python.exe -m langgraph dev  # 启动 LangGraph
pytest tests/ -v                            # 运行测试

# 前端开发
cd frontend
pnpm dev                                    # 启动开发服务器
pnpm typecheck                              # 类型检查
pnpm lint                                   # 代码检查
pnpm lint:fix                               # 修复代码格式

# 生产构建
$env:BETTER_AUTH_SECRET = "your-secret"
pnpm build                                  # 生产构建

# 查看日志
Get-Content logs/langgraph.log
Get-Content logs/gateway.log
